let active = false, distance = 0, fare = 0, waitSeconds = 0; let geoWatch, lastPos, liveInterval, startTime, endTime; const params = new URLSearchParams(window.location.search); const brand = params.get('brand') || 'get'; const tType = params.get('t') || '1'; document.getElementById('logo-img').src = brand === 'red' ? "red-taxi.png" : "get-taxi.png"; let config = { base: 80, rate: 25 }; if(tType === '2') config = { base: 100, rate: 25 }; if(tType === '3') config = { base: 80, rate: 27 }; function toggleTrip() { if(!active) { // Pre-warm audio for mobile browsers const wakeUp = new Audio(); wakeUp.play().catch(() => {}); active = true; fare = config.base; startTime = new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}); document.getElementById('mainBtn').innerText = "Stop Trip"; startTracking(); liveInterval = setInterval(updateLiveWaiting, 1000); } else { finishTrip(); } } function updateLiveWaiting() { // Check if car is stopped if (!lastPos || (lastPos.coords.speed || 0) < 0.6) { waitSeconds++; // ADD ₹2 ONLY WHEN A FULL MINUTE (60 SECONDS) IS COMPLETED if (waitSeconds > 0 && waitSeconds % 60 === 0) { fare += 2; } document.getElementById('status-label').innerText = "Waiting..."; } else { document.getElementById('status-label').innerText = "Driving"; } updateUI(); } function finishTrip() { active = false; clearInterval(liveInterval); endTime = new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}); navigator.geolocation.clearWatch(geoWatch); // VOICE SEQUENCE: Start.mp3 -> Fare -> end.mp3 let audioStart = new Audio('Start.mp3'); audioStart.play().then(() => { audioStart.onended = function() { let amountSpeech = new SpeechSynthesisUtterance(Math.floor(fare) + " Rupees"); amountSpeech.lang = 'en-US'; amountSpeech.rate = 0.85; window.speechSynthesis.speak(amountSpeech); amountSpeech.onend = function() { let audioEnd = new Audio('end.mp3'); audioEnd.play(); }; }; }).catch(e => { console.log("Audio play blocked. Total: " + fare); alert("Total Fare: ₹" + fare.toFixed(2)); }); document.getElementById('live-meter').style.display = 'none'; document.getElementById('receipt-view').style.display = 'flex'; document.getElementById('r-start').innerText = startTime; document.getElementById('r-end').innerText = endTime; document.getElementById('r-dist').innerText = distance.toFixed(2) + ' KM'; let mins = Math.floor(waitSeconds / 60); let secs = waitSeconds % 60; document.getElementById('r-wait').innerText = mins + ":" + (secs < 10 ? "0" : "") + secs; document.getElementById('r-total').innerText = '₹' + fare.toFixed(2); } function shareTripDetails() { let mins = Math.floor(waitSeconds / 60); let secs = waitSeconds % 60; let waitTimeStr = mins + ":" + (secs < 10 ? "0" : "") + secs; let msg = `*TRIP DETAILS*%0A*Pickup:* ${startTime}%0A*Drop:* ${endTime}%0A*KM:* ${distance.toFixed(2)}%0A*Wait:* ${waitTimeStr}%0A*TOTAL: ₹${fare.toFixed(2)}*`; window.open(`https://wa.me/919043743777?text=${msg}`, '_blank'); } function startTracking() { geoWatch = navigator.geolocation.watchPosition(p => { if(lastPos && (p.coords.speed || 0) >= 0.6) { let d = calcDist(lastPos.coords.latitude, lastPos.coords.longitude, p.coords.latitude, p.coords.longitude); if (d > 0.005) { distance += d; fare += (d * config.rate); // Driving fare increases LIVE } } lastPos = p; document.getElementById('speed').innerText = (p.coords.speed * 3.6 || 0).toFixed(1); updateUI(); }, null, {enableHighAccuracy: true}); } function calcDist(la1, lo1, la2, lo2) { const R = 6371; const dLa = (la2-la1)*Math.PI/180; const dLo = (lo2-lo1)*Math.PI/180; const a = Math.sin(dLa/2)**2 + Math.cos(la1*Math.PI/180)*Math.cos(la2*Math.PI/180)*Math.sin(dLo/2)**2; return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); } function updateUI() { let parts = fare.toFixed(2).split('.'); document.getElementById('fare-whole').innerText = parts[0]; document.getElementById('fare-dec').innerText = parts[1]; document.getElementById('dist').innerText = distance.toFixed(2); let mins = Math.floor(waitSeconds / 60); let secs = waitSeconds % 60; document.getElementById('wait-display').innerText = mins + ":" + (secs < 10 ? "0" : "") + secs; }
